GraphQL Query

```javascript
query ValidateDevice($policy: DevicePolicy!) {
  policy {
    validate(policy: $policy) {
      status
      osVersion
      firewall
      diskEncryption
      stethoscopeVersion
    }
  }

  device {
    deviceId
    deviceName
    platform
    platformName
    friendlyName
    osVersion
    osName
    osBuild
    firmwareVersion
    hardwareModel
    hardwareSerial
    stethoscopeVersion
    osqueryVersion
    ipAddresses {
      interface
      address
      mask
      broadcast
    }
    macAddresses {
      interface
      type
      mac
      physicalAdapter
      lastChange
    }
    security {
      firewall
      diskEncryption
      automaticAppUpdates
      automaticSecurityUpdates
      automaticOsUpdates
    }
  }
}
```
